46  /**
47   * A concrete implementation of the {@link IXdmFactory} used to create various {@link INodeItem}
48   * instances for use in evaluating Metapath expressions.
49   */
50  public class DefaultXdmFactory implements IXdmFactory {
51  
52    @Override
53    public IDocumentNodeItem newDocumentNodeItem(RootAssemblyDefinition definition, Object value,
54        URI documentUri) {
55      return new XdmDocumentNodeItemImpl(definition, value, documentUri);
56    }
57  
58    @Override
59    public IAssemblyNodeItem newAssemblyNodeItem(IBoundAssemblyInstance instance, Object value, int position,
60        IAssemblyNodeItem parentNodeItem) {
61      return new IntermediateXdmAssemblyNodeItemImpl(instance, value, position, parentNodeItem);
62    }
63  
64    @Override
65    public IAssemblyNodeItem newAssemblyNodeItem(IAssemblyClassBinding definition, Object value,
66        URI baseUri) {
67      return new OrphanedXdmAssemblyNodeItemImpl(definition, value, 1, baseUri);
68    }
69  
70    @Override
71    public IFieldNodeItem newFieldNodeItem(IBoundFieldInstance instance, Object value, int position,
72        IAssemblyNodeItem parentNodeItem) {
73      return new IntermediateXdmFieldNodeItemImpl(instance, value, position, parentNodeItem);
74    }
75  
76    @Override
77    public IFieldNodeItem newFieldNodeItem(IBoundFieldDefinition definition, Object value,
78        URI baseUri) {
79      return new OrphanedXdmFieldNodeItemImpl(definition, value, 1, baseUri);
80    }
81  
82    @Override
83    public IFlagNodeItem newFlagNodeItem(IBoundFlagInstance instance, Object value,
84        IModelNodeItem parentNodeItem) {
85      return new XdmFlagNodeItemImpl(instance, value, parentNodeItem);
86    }
87  
88    @Override
89    public INodeItem newNodeItem(IClassBinding definition, Object value, URI baseUri, boolean rootNode) {
90      INodeItem retval;
91      if (definition instanceof IAssemblyClassBinding) {
92        if (rootNode) {
93          retval = newDocumentNodeItem(new RootAssemblyDefinition((IAssemblyClassBinding) definition), value, baseUri);
94        } else {
95          retval = newAssemblyNodeItem((IAssemblyClassBinding) definition, value, baseUri);
96        }
97      } else if (definition instanceof IFieldClassBinding) {
98        retval = newFieldNodeItem((IFieldClassBinding) definition, value, baseUri);
99      } else {
100       throw new UnsupportedOperationException("must be a bound assembly or field");
101     }
102     return retval;
103   }
104 }
