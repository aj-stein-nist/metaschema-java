111   @NotNull

116   @Override
117   public ISequence<? extends IBooleanItem> visitAnd(And expr, INodeContext context) {
118     boolean retval = true;
119     for (IExpression child : expr.getChildren()) {
120       ISequence<?> result = child.accept(this, context);
121       if (!XPathFunctions.fnBooleanAsPrimative(result)) {
122         retval = false;
123         break;
124       }
125     }
126     return ISequence.of(IBooleanItem.valueOf(retval));
127   }
128 
129   @Override
130   public ISequence<? extends IBooleanItem> visitOr(Or expr, INodeContext context) {
131     boolean retval = false;
132     for (IExpression child : expr.getChildren()) {
133       ISequence<?> result = child.accept(this, context);
134       if (XPathFunctions.fnBooleanAsPrimative(result)) {
135         retval = true;
136         break;
137       }
138     }
139     return ISequence.of(IBooleanItem.valueOf(retval));
140   }

142   @Override
143   public ISequence<? extends IBooleanItem> visitValueComparison(ValueComparison expr, INodeContext context) {
144     IItem leftItem = FunctionUtils.getFirstItem(expr.getLeft().accept(this, context), false);
145     if (leftItem == null) {
146       return ISequence.empty();
147     }
148     IItem rightItem = FunctionUtils.getFirstItem(expr.getRight().accept(this, context), false);
149     if (rightItem == null) {
150       return ISequence.empty();
151     }
152 
153     IAnyAtomicItem left = XPathFunctions.fnDataItem(leftItem);
154     IAnyAtomicItem right = XPathFunctions.fnDataItem(rightItem);
155     IComparison.Operator operator = expr.getOperator();
156 
157     try {
158       return resultOrEmptySequence(compare(left, operator, right));
159     } catch (IllegalArgumentException ex) {
160       throw new UnsupportedOperationException(String.format("The value expression '%s %s %s' is not supported",
161           leftItem.getItemName(), operator.name().toLowerCase(), rightItem.getItemName()));
162     }
163   }

165   protected IBooleanItem compare(@NotNull IAnyAtomicItem left, @NotNull IComparison.Operator operator,
166       @NotNull IAnyAtomicItem right) {
167     IBooleanItem retval = null;
168     if (left instanceof IStringItem || right instanceof IStringItem) {
169       switch (operator) {
170       case EQ:
171         retval = OperationFunctions.opNumericEqual(XPathFunctions.fnCompare((IStringItem) left, (IStringItem) right),
172             IIntegerItem.ZERO);
173         break;
174       case GE:
175         retval = OperationFunctions.opNumericGreaterThan(
176             XPathFunctions.fnCompare((IStringItem) left, (IStringItem) right), IIntegerItem.NEGATIVE_ONE);
177         break;
178       case GT:
179         retval = OperationFunctions
180             .opNumericGreaterThan(XPathFunctions.fnCompare((IStringItem) left, (IStringItem) right), IIntegerItem.ZERO);
181         break;
182       case LE:
183         retval = OperationFunctions.opNumericLessThan(XPathFunctions.fnCompare((IStringItem) left, (IStringItem) right),
184             IIntegerItem.ONE);
185         break;
186       case LT:
187         retval = OperationFunctions.opNumericLessThan(XPathFunctions.fnCompare((IStringItem) left, (IStringItem) right),
188             IIntegerItem.ZERO);
189         break;
190       case NE:
191         retval = FnNotFunction.fnNot(OperationFunctions
192             .opNumericEqual(XPathFunctions.fnCompare((IStringItem) left, (IStringItem) right), IIntegerItem.ZERO));
193         break;
194       default:
195         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
196       }
197     } else if (left instanceof INumericItem && right instanceof INumericItem) {
198       switch (operator) {
199       case EQ:
200         retval = OperationFunctions.opNumericEqual((INumericItem) left, (INumericItem) right);
201         break;
202       case GE: {
203         IBooleanItem gt = OperationFunctions.opNumericGreaterThan((INumericItem) left, (INumericItem) right);
204         IBooleanItem eq = OperationFunctions.opNumericEqual((INumericItem) left, (INumericItem) right);
205         retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
206         break;
207       }
208       case GT:
209         retval = OperationFunctions.opNumericGreaterThan((INumericItem) left, (INumericItem) right);
210         break;
211       case LE: {
212         IBooleanItem lt = OperationFunctions.opNumericLessThan((INumericItem) left, (INumericItem) right);
213         IBooleanItem eq = OperationFunctions.opNumericEqual((INumericItem) left, (INumericItem) right);
214         retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
215         break;
216       }
217       case LT:
218         retval = OperationFunctions.opNumericLessThan((INumericItem) left, (INumericItem) right);
219         break;
220       case NE:
221         retval = FnNotFunction.fnNot(OperationFunctions.opNumericEqual((INumericItem) left, (INumericItem) right));
222         break;
223       default:
224         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
225       }
226     } else if (left instanceof IBooleanItem && right instanceof IBooleanItem) {
227       switch (operator) {
228       case EQ:
229         retval = OperationFunctions.opBooleanEqual((IBooleanItem) left, (IBooleanItem) right);
230         break;
231       case GE: {
232         IBooleanItem gt = OperationFunctions.opBooleanGreaterThan((IBooleanItem) left, (IBooleanItem) right);
233         IBooleanItem eq = OperationFunctions.opBooleanEqual((IBooleanItem) left, (IBooleanItem) right);
234         retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
235         break;
236       }
237       case GT:
238         retval = OperationFunctions.opBooleanGreaterThan((IBooleanItem) left, (IBooleanItem) right);
239         break;
240       case LE: {
241         IBooleanItem lt = OperationFunctions.opBooleanLessThan((IBooleanItem) left, (IBooleanItem) right);
242         IBooleanItem eq = OperationFunctions.opBooleanEqual((IBooleanItem) left, (IBooleanItem) right);
243         retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
244         break;
245       }
246       case LT:
247         retval = OperationFunctions.opBooleanLessThan((IBooleanItem) left, (IBooleanItem) right);
248         break;
249       case NE:
250         retval = FnNotFunction.fnNot(OperationFunctions.opBooleanEqual((IBooleanItem) left, (IBooleanItem) right));
251         break;
252       default:
253         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
254       }
255     } else if (left instanceof IDateTimeItem && right instanceof IDateTimeItem) {
256       switch (operator) {
257       case EQ:
258         retval = OperationFunctions.opDateTimeEqual((IDateTimeItem) left, (IDateTimeItem) right);
259         break;
260       case GE: {
261         IBooleanItem gt = OperationFunctions.opDateTimeGreaterThan((IDateTimeItem) left, (IDateTimeItem) right);
262         IBooleanItem eq = OperationFunctions.opDateTimeEqual((IDateTimeItem) left, (IDateTimeItem) right);
263         retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
264         break;
265       }
266       case GT:
267         retval = OperationFunctions.opDateTimeGreaterThan((IDateTimeItem) left, (IDateTimeItem) right);
268         break;
269       case LE: {
270         IBooleanItem lt = OperationFunctions.opDateTimeLessThan((IDateTimeItem) left, (IDateTimeItem) right);
271         IBooleanItem eq = OperationFunctions.opDateTimeEqual((IDateTimeItem) left, (IDateTimeItem) right);
272         retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
273         break;
274       }
275       case LT:
276         retval = OperationFunctions.opDateTimeLessThan((IDateTimeItem) left, (IDateTimeItem) right);
277         break;
278       case NE:
279         retval = FnNotFunction.fnNot(OperationFunctions.opDateTimeEqual((IDateTimeItem) left, (IDateTimeItem) right));
280         break;
281       default:
282         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
283       }
284     } else if (left instanceof IDateItem && right instanceof IDateItem) {
285       switch (operator) {
286       case EQ:
287         retval = OperationFunctions.opDateEqual((IDateItem) left, (IDateItem) right);
288         break;
289       case GE: {
290         IBooleanItem gt = OperationFunctions.opDateGreaterThan((IDateItem) left, (IDateItem) right);
291         IBooleanItem eq = OperationFunctions.opDateEqual((IDateItem) left, (IDateItem) right);
292         retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
293         break;
294       }
295       case GT:
296         retval = OperationFunctions.opDateGreaterThan((IDateItem) left, (IDateItem) right);
297         break;
298       case LE: {
299         IBooleanItem lt = OperationFunctions.opDateLessThan((IDateItem) left, (IDateItem) right);
300         IBooleanItem eq = OperationFunctions.opDateEqual((IDateItem) left, (IDateItem) right);
301         retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
302         break;
303       }
304       case LT:
305         retval = OperationFunctions.opDateLessThan((IDateItem) left, (IDateItem) right);
306         break;
307       case NE:
308         retval = FnNotFunction.fnNot(OperationFunctions.opDateEqual((IDateItem) left, (IDateItem) right));
309         break;
310       default:
311         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
312       }
313     } else if (left instanceof IDurationItem && right instanceof IDurationItem) {
314       switch (operator) {
315       case EQ:
316         retval = OperationFunctions.opDurationEqual((IDurationItem) left, (IDurationItem) right);
317         break;
318       case GE: {
319         if (left instanceof IYearMonthDurationItem && right instanceof IYearMonthDurationItem) {
320           IBooleanItem gt = OperationFunctions.opYearMonthDurationGreaterThan((IYearMonthDurationItem) left,
321               (IYearMonthDurationItem) right);
322           IBooleanItem eq
323               = OperationFunctions.opDurationEqual((IYearMonthDurationItem) left, (IYearMonthDurationItem) right);
324           retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
325         } else if (left instanceof IDayTimeDurationItem && right instanceof IDayTimeDurationItem) {
326           IBooleanItem gt = OperationFunctions.opDayTimeDurationGreaterThan((IDayTimeDurationItem) left,
327               (IDayTimeDurationItem) right);
328           IBooleanItem eq
329               = OperationFunctions.opDurationEqual((IDayTimeDurationItem) left, (IDayTimeDurationItem) right);
330           retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
331         } else {
332           throw new IllegalArgumentException("the item types are not comparable");
333         }
334         break;
335       }
336       case GT:
337         if (left instanceof IYearMonthDurationItem && right instanceof IYearMonthDurationItem) {
338           retval = OperationFunctions.opYearMonthDurationGreaterThan((IYearMonthDurationItem) left,
339               (IYearMonthDurationItem) right);
340         } else if (left instanceof IDayTimeDurationItem && right instanceof IDayTimeDurationItem) {
341           retval = OperationFunctions.opDayTimeDurationGreaterThan((IDayTimeDurationItem) left,
342               (IDayTimeDurationItem) right);
343         } else {
344           throw new IllegalArgumentException("the item types are not comparable");
345         }
346         break;
347       case LE: {
348         if (left instanceof IYearMonthDurationItem && right instanceof IYearMonthDurationItem) {
349           IBooleanItem lt = OperationFunctions.opYearMonthDurationLessThan((IYearMonthDurationItem) left,
350               (IYearMonthDurationItem) right);
351           IBooleanItem eq
352               = OperationFunctions.opDurationEqual((IYearMonthDurationItem) left, (IYearMonthDurationItem) right);
353           retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
354         } else if (left instanceof IDayTimeDurationItem && right instanceof IDayTimeDurationItem) {
355           IBooleanItem lt
356               = OperationFunctions.opDayTimeDurationLessThan((IDayTimeDurationItem) left, (IDayTimeDurationItem) right);
357           IBooleanItem eq
358               = OperationFunctions.opDurationEqual((IDayTimeDurationItem) left, (IDayTimeDurationItem) right);
359           retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
360         } else {
361           throw new IllegalArgumentException("the item types are not comparable");
362         }
363         break;
364       }
365       case LT: {
366         if (left instanceof IYearMonthDurationItem && right instanceof IYearMonthDurationItem) {
367           retval = OperationFunctions.opYearMonthDurationLessThan((IYearMonthDurationItem) left,
368               (IYearMonthDurationItem) right);
369         } else if (left instanceof IDayTimeDurationItem && right instanceof IDayTimeDurationItem) {
370           retval
371               = OperationFunctions.opDayTimeDurationLessThan((IDayTimeDurationItem) left, (IDayTimeDurationItem) right);
372         } else {
373           throw new IllegalArgumentException("the item types are not comparable");
374         }
375         break;
376       }
377       case NE:
378         retval = FnNotFunction.fnNot(OperationFunctions.opDurationEqual((IDurationItem) left, (IDurationItem) right));
379         break;
380       default:
381         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
382       }
383     } else if (left instanceof IBase64BinaryItem && right instanceof IBase64BinaryItem) {
384       switch (operator) {
385       case EQ:
386         retval = OperationFunctions.opBase64BinaryEqual((IBase64BinaryItem) left, (IBase64BinaryItem) right);
387         break;
388       case GE: {
389         IBooleanItem gt
390             = OperationFunctions.opBase64BinaryGreaterThan((IBase64BinaryItem) left, (IBase64BinaryItem) right);
391         IBooleanItem eq = OperationFunctions.opBase64BinaryEqual((IBase64BinaryItem) left, (IBase64BinaryItem) right);
392         retval = IBooleanItem.valueOf(gt.toBoolean() || eq.toBoolean());
393         break;
394       }
395       case GT:
396         retval = OperationFunctions.opBase64BinaryGreaterThan((IBase64BinaryItem) left, (IBase64BinaryItem) right);
397         break;
398       case LE: {
399         IBooleanItem lt
400             = OperationFunctions.opBase64BinaryLessThan((IBase64BinaryItem) left, (IBase64BinaryItem) right);
401         IBooleanItem eq = OperationFunctions.opBase64BinaryEqual((IBase64BinaryItem) left, (IBase64BinaryItem) right);
402         retval = IBooleanItem.valueOf(lt.toBoolean() || eq.toBoolean());
403         break;
404       }
405       case LT:
406         retval = OperationFunctions.opBase64BinaryLessThan((IBase64BinaryItem) left, (IBase64BinaryItem) right);
407         break;
408       case NE:
409         retval = FnNotFunction
410             .fnNot(OperationFunctions.opBase64BinaryEqual((IBase64BinaryItem) left, (IBase64BinaryItem) right));
411         break;
412       default:
413         throw new IllegalArgumentException(String.format("Unsupported operator '%s'", operator.name()));
414       }
415     } else {
416       throw new InvalidTypeMetapathException(String.format("invalid types for comparison: %s %s %s", left.getItemName(),
417           operator.name().toLowerCase(), right.getItemName()));
418     }
419 
420     return retval;
421   }

423   @Override
424   public @NotNull ISequence<? extends IBooleanItem> visitGeneralComparison(@NotNull GeneralComparison expr,
425       @NotNull INodeContext context) {
426     ISequence<? extends IAnyAtomicItem> leftItems = XPathFunctions.fnData(expr.getLeft().accept(this, context));
427     ISequence<? extends IAnyAtomicItem> rightItems = XPathFunctions.fnData(expr.getRight().accept(this, context));
428     IComparison.Operator operator = expr.getOperator();
429 
430     IBooleanItem retval = IBooleanItem.FALSE;
431     // TODO: apply data on each iteration
432     for (IAnyAtomicItem left : leftItems.asList()) {
433       for (IAnyAtomicItem right : rightItems.asList()) {
434         @NotNull
435         IAnyAtomicItem leftCast;
436         IAnyAtomicItem rightCast;
437         if (left instanceof IUntypedAtomicItem) {
438           if (right instanceof IUntypedAtomicItem) {
439             leftCast = IStringItem.cast(left);
440             rightCast = IStringItem.cast(right);
441           } else {
442             leftCast = applyGeneralComparisonCast(right, left);
443             rightCast = right;
444           }
445         } else if (right instanceof IUntypedAtomicItem) {
446           leftCast = left;
447           rightCast = applyGeneralComparisonCast(left, right);
448         } else {
449           leftCast = left;
450           rightCast = right;
451         }
452 
453         try {
454           IBooleanItem result = compare(leftCast, operator, rightCast);
455           if (IBooleanItem.TRUE.equals(result)) {
456             retval = IBooleanItem.TRUE;
457           }
458         } catch (IllegalArgumentException ex) {
459           throw new UnsupportedOperationException(String.format("The value expression '%s %s %s' is not supported",
460               left.getItemName(), operator.name().toLowerCase(), right.getItemName()));
461         }
462       }
463     }
464     return ISequence.of(retval);
465   }
466 
467   @NotNull
468   protected IAnyAtomicItem applyGeneralComparisonCast(@NotNull IAnyAtomicItem item, @NotNull IAnyAtomicItem other) {
469     IAnyAtomicItem retval = other;
470     if (item instanceof INumericItem) {
471       retval = IDecimalItem.cast(other);
472     } else if (item instanceof IDayTimeDurationItem) {
473       retval = IDayTimeDurationItem.cast(other);
474     } else if (item instanceof IDayTimeDurationItem) {
475       retval = IYearMonthDurationItem.cast(other);
476     } else {
477       retval = item.getJavaTypeAdapter().cast(other);
478     }
479     return retval;
480   }

530   @Override
531   public ISequence<?> visitStep(Step expr, INodeContext context) {
532 
533     ISequence<?> stepResult = expr.getStep().accept(this, context);
534 
535     // evaluate the predicates for this step
536     AtomicInteger index = new AtomicInteger();
537 
538     Stream<? extends IItem> stream = stepResult.asStream().map(item -> {
539       // build a positional index of the items
540       return Map.entry(BigInteger.valueOf(index.incrementAndGet()), item);
541     }).filter(entry -> {
542       @SuppressWarnings("null")
543       @NotNull
544       IItem item = entry.getValue();
545 
546       // return false if any predicate evaluates to false
547       boolean result = !expr.getPredicates().stream().map(predicateExpr -> {
548         boolean bool;
549         if (predicateExpr instanceof IntegerLiteral) {
550           // reduce the result to the matching item
551           BigInteger predicateIndex = ((IntegerLiteral) predicateExpr).getValue();
552 
553           // get the position of the item
554           final BigInteger position = entry.getKey();
555 
556           // it is a match if the position matches
557           bool = position.equals(predicateIndex);
558         } else {
559           INodeContext childContext = (INodeContext) item;
560           ISequence<?> predicateResult = predicateExpr.accept(this, childContext);
561           bool = XPathFunctions.fnBoolean(predicateResult).toBoolean();
562         }
563         return bool;
564       }).anyMatch(x -> !x);
565       return result;
566     }).map(entry -> entry.getValue());
567     @SuppressWarnings("null")
568     ISequence<?> retval = ISequence.of(stream);
569     return retval;
570   }

656   protected Stream<? extends INodeItem> searchExpression(@NotNull IExpression expr, @NotNull INodeContext context) {
657 
658     // check the current node
659     @SuppressWarnings("unchecked")
660     Stream<? extends INodeItem> nodeMatches
661         = (Stream<? extends INodeItem>) expr.accept(this, context).asStream();
662 
663     // create a stream of flags and model elements to check
664     Stream<? extends IFlagNodeItem> flags = context.flags();
665     Stream<? extends INodeItem> modelItems = context.modelItems();
666 
667     @SuppressWarnings("null")
668     Stream<? extends INodeItem> childMatches = Stream.concat(flags, modelItems)
669         .flatMap(instance -> {
670           return searchExpression(expr, instance);
671         });
672

674     @NotNull
675     Stream<? extends INodeItem> result = Stream.concat(nodeMatches, childMatches);
676     return result;

689   @NotNull
690   protected Stream<? extends IModelNodeItem> searchModelInstances(@NotNull ModelInstance modelInstance,
691       @NotNull INodeContext context) {
692 
693     // check if the current node context matches the expression
694     Stream<? extends IModelNodeItem> nodeMatches = matchModelInstance(modelInstance, context);

696     // next iterate over the child model instances, if the context item is an assembly
697     @SuppressWarnings("null")
698     Stream<? extends IModelNodeItem> childMatches
699         = context.modelItems().flatMap(modelItem -> {
700           // apply the search criteria to these node items
701           return searchModelInstances(modelInstance, modelItem);
702         });
703 
704     // combine the results
705     @SuppressWarnings("null")
706     @NotNull
707     Stream<? extends IModelNodeItem> retval = Stream.concat(nodeMatches, childMatches);
708     return retval;
709   }
