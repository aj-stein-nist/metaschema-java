68    /**
69     * Generate Java sources for the provided metaschema.
70     * 
71     * @param metaschema
72     *          the Metaschema to generate Java sources for
73     * @param targetDir
74     *          the directory to generate sources in
75     * @param bindingConfiguration
76     *          the binding customizations to use when generating the Java classes
77     * @return a mapping of a list of generated classes (value) for a given Metaschema instance (key)
78     * @throws IOException
79     *           if a build error occurred while generating the class
80     */
81    public static Map<IMetaschema, List<GeneratedClass>> generate(IMetaschema metaschema, File targetDir,
82        IBindingConfiguration bindingConfiguration) throws IOException {
83      return generate(Collections.singletonList(metaschema), targetDir, bindingConfiguration);
84    }

86    /**
87     * Generates Java classes for Metaschema fields and flags.
88     * 
89     * @param metaschemas
90     *          the Metaschema instances to build classes for
91     * @param targetDirectory
92     *          the directory to generate classes in
93     * @param bindingConfiguration
94     *          binding customizations that can be used to set namespaces, class names, and other
95     *          aspects of generated classes
96     * @return a mapping of a list of generated classes (value) for a given Metaschema instance (key)
97     * @throws IOException
98     *           if a build error occurred while generating the class
99     */
100   public static Map<IMetaschema, List<GeneratedClass>> generate(Collection<? extends IMetaschema> metaschemas,
101       File targetDirectory, IBindingConfiguration bindingConfiguration) throws IOException {
102     Objects.requireNonNull(metaschemas, "metaschemas");
103     Objects.requireNonNull(targetDirectory, "generationTargetDirectory");
104     Objects.requireNonNull(bindingConfiguration, "bindingConfiguration");
105     if (LOGGER.isInfoEnabled()) {
106       LOGGER.info("Generating Java classes in: {}", targetDirectory.getPath());
107     }
108 
109     Map<URI, String> xmlNamespaceToPackageNameMap = new HashMap<>();
110     Map<URI, Set<IMetaschema>> xmlNamespaceToMetaschemaMap = new HashMap<>();
111 
112     ITypeResolver typeResolver = new DefaultTypeResolver(bindingConfiguration);
113 
114     Map<IMetaschema, List<? extends IDefinition>> metaschemaToInformationElementsMap
115         = buildMetaschemaMap(metaschemas);
116 
117     Map<IMetaschema, List<GeneratedClass>> retval = new HashMap<>();
118     for (Map.Entry<IMetaschema, List<? extends IDefinition>> entry : metaschemaToInformationElementsMap
119         .entrySet()) {
120       IMetaschema metaschema = entry.getKey();
121       List<GeneratedClass> generatedClasses = null;
122       Set<String> classNames = new HashSet<>();
123 
124       for (IDefinition definition : entry.getValue()) {
125         IJavaClassGenerator classGenerator = null;
126         if (definition instanceof IAssemblyDefinition) {
127           classGenerator = new AssemblyJavaClassGenerator((IAssemblyDefinition) definition, typeResolver);
128         } else if (definition instanceof IFieldDefinition) {
129           IFieldDefinition fieldDefinition = (IFieldDefinition) definition;
130 
131           if (fieldDefinition.getFlagInstances().isEmpty()) {
132             // if field is just a simple data value, then no class is needed
133           } else {
134             classGenerator = new FieldJavaClassGenerator(fieldDefinition, typeResolver);
135           }
136         } else {
137           // Skip others
138           continue;
139         }
140 
141         if (classGenerator != null) {
142           GeneratedClass generatedClass;
143           try {
144             generatedClass = classGenerator.generateClass(targetDirectory);
145           } catch (RuntimeException ex) {
146             throw new IllegalStateException(
147                 String.format("Unable to generate class for definition '%s' in Metaschema '%s'", definition.getName(),
148                     metaschema.getLocation()),
149                 ex);
150           }
151           String className = generatedClass.getClassName().canonicalName();
152           if (classNames.contains(className)) {
153             throw new IllegalStateException(String.format(
154                 "Found duplicate class name '%s' in metaschema '%s'."
155                     + " If multiple metaschema are compiled for the same namespace, all class names must be unique.",
156                 className, metaschema.getLocation()));
157           }
158           classNames.add(className);
159 
160           if (generatedClasses == null) {
161             generatedClasses = new LinkedList<>();
162             retval.put(metaschema, generatedClasses);
163           }
164           generatedClasses.add(generatedClass);
165 
166         }
167       }
168 
169       URI xmlNamespace = metaschema.getXmlNamespace();
170       String packageName = bindingConfiguration.getPackageNameForMetaschema(metaschema);
171 
172       if (xmlNamespaceToPackageNameMap.containsKey(xmlNamespace)) {
173         String assignedPackage = xmlNamespaceToPackageNameMap.get(xmlNamespace);
174         if (!assignedPackage.equals(packageName)) {
175           throw new IllegalStateException(String.format(
176               "The metaschema '%s' is assigning the new package name '%s'."
177                   + " This new name is different than the previously assigned package name '%s' for the same namespace."
178                   + " A metaschema namespace must be assigned a consistent package name.",
179               metaschema.getLocation().toString(), packageName, assignedPackage));
180         }
181       } else {
182         xmlNamespaceToPackageNameMap.put(xmlNamespace, packageName);
183       }
184 
185       Set<IMetaschema> metaschemaSet = xmlNamespaceToMetaschemaMap.get(xmlNamespace);
186       if (metaschemaSet == null) {
187         metaschemaSet = new HashSet<>();
188         xmlNamespaceToMetaschemaMap.put(xmlNamespace, metaschemaSet);
189       }
190       metaschemaSet.add(metaschema);
191     }
192 
193     for (Map.Entry<URI, String> entry : xmlNamespaceToPackageNameMap.entrySet()) {
194       String packageName = entry.getValue();
195       String packagePath = packageName.replace(".", "/");
196       File packageInfo = new File(targetDirectory, packagePath + "/package-info.java");
197       URI namespace = entry.getKey();
198       String namespaceString = namespace.toString();
199 
200       try (PrintWriter writer = new PrintWriter(
201           Files.newBufferedWriter(packageInfo.toPath(), StandardOpenOption.CREATE, StandardOpenOption.WRITE))) {
202         writer.format(
203             "@%1$s(namespace = \"%2$s\", xmlns = {@%3$s(prefix = \"\", namespace = \"%2$s\")},"
204                 + " xmlElementFormDefault = %4$s.QUALIFIED)%n",
205             XmlSchema.class.getName(), namespaceString, XmlNs.class.getName(), XmlNsForm.class.getName());
206         writer.format("package %s;%n", packageName);
207       }
208 
209       for (IMetaschema metaschema : xmlNamespaceToMetaschemaMap.get(namespace)) {
210         GeneratedClass packageInfoClass
211             = new GeneratedClass(packageInfo, ClassName.get(packageName, "package-info"), false);
212         List<GeneratedClass> generatedClasses = retval.get(metaschema);
213         if (generatedClasses == null) {
214           generatedClasses = Collections.singletonList(packageInfoClass);
215         } else {
216           generatedClasses.add(packageInfoClass);
217         }
218       }
219     }
220     return Collections.unmodifiableMap(retval);
221   }
222 
223   private static Map<IMetaschema, List<? extends IDefinition>>
224       buildMetaschemaMap(Collection<? extends IMetaschema> metaschemas) {
225     Map<IMetaschema, List<? extends IDefinition>> retval = new HashMap<>();
226 
227     for (IMetaschema metaschema : metaschemas) {
228       processMetaschema(metaschema, retval);
229     }
230     return retval;
231   }
232 
233   private static void processMetaschema(IMetaschema metaschema,
234       Map<IMetaschema, List<? extends IDefinition>> map) {
235     for (IMetaschema importedMetaschema : metaschema.getImportedMetaschemas()) {
236       processMetaschema(importedMetaschema, map);
237     }
238 
239     if (!map.containsKey(metaschema)) {
240       List<? extends IDefinition> definitions = metaschema.getAssemblyAndFieldDefinitions();
241       map.put(metaschema, definitions);
242     }
243   }

245   public static class GeneratedClass {
246     private final File classFile;
247     private final ClassName className;
248     private final boolean rootClass;
249 
250     /**
251      * Construct a new class information object for a generated class.
252      * 
253      * @param classFile
254      *          the file the class was written to
255      * @param className
256      *          the type info for the class
257      * @param rootClass
258      *          {@code true} if the class is a root assembly, or {@code false} otherwise
259      */
260     public GeneratedClass(File classFile, ClassName className, boolean rootClass) {
261       Objects.requireNonNull(classFile, "classFile");
262       Objects.requireNonNull(className, "className");
263       this.classFile = classFile;
264       this.className = className;
265       this.rootClass = rootClass;
266     }
267 
268     /**
269      * The file the class was written to.
270      * 
271      * @return the class file
272      */
273     public File getClassFile() {
274       return classFile;
275     }
276 
277     /**
278      * The type info for the class.
279      * 
280      * @return the class's type info
281      */
282     public ClassName getClassName() {
283       return className;
284     }
285 
286     /**
287      * Indicates if the class represents a root Metaschema assembly which can be the top-level
288      * element/property of an XML, JSON, or YAML instance.
289      * 
290      * @return {@code true} if the class is a root assembly, or {@code false} otherwise
291      */
292     public boolean isRootClass() {
293       return rootClass;
294     }
295   }
