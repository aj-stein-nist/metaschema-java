61  final class ConstraintFactory {
62    private ConstraintFactory() {
63      // disable
64    }
65  
66    static Map<String, DefaultAllowedValue> toAllowedValues(AllowedValues constraint) {
67      AllowedValue[] values = constraint.values();
68  
69      Map<String, DefaultAllowedValue> allowedValues = new LinkedHashMap<>(values.length);
70      for (AllowedValue value : values) {
71        DefaultAllowedValue allowedValue
72            = new DefaultAllowedValue(value.value(), MarkupLine.fromMarkdown(value.description()));
73        allowedValues.put(allowedValue.getValue(), allowedValue);
74      }
75      return Collections.unmodifiableMap(allowedValues);
76    }
77  
78    static String toId(String id) {
79      return id.isBlank() ? null : id;
80    }
81  
82    static MarkupMultiline toRemarks(String remarks) {
83      return remarks.isBlank() ? null : MarkupMultiline.fromMarkdown(remarks);
84    }
85  
86    static Pattern toPattern(String pattern) {
87      return pattern.isBlank() ? null : Pattern.compile(pattern);
88    }
89  
90    static String toMessage(String message) {
91      return message.isBlank() ? null : message;
92    }
93  
94    static IJavaTypeAdapter<?> toDataType(Class<? extends IJavaTypeAdapter<?>> adapterClass) {
95      return adapterClass.isAssignableFrom(NullJavaTypeAdapter.class) ? null
96          : DataTypeService.getInstance().getJavaTypeAdapterByClass(adapterClass);
97    }
98  
99    static MetapathExpression toMetapath(String metapath) {
100     return metapath == null || metapath.isBlank() ? null : MetapathExpression.compile(metapath);
101   }
102 
103   static DefaultAllowedValuesConstraint newAllowedValuesConstraint(AllowedValues constraint) {
104     return new DefaultAllowedValuesConstraint(
105         toId(constraint.id()),
106         constraint.level(),
107         toMetapath(constraint.target()),
108         toAllowedValues(constraint),
109         constraint.allowOthers(),
110         toRemarks(constraint.remarks()));
111   }
112 
113   static DefaultMatchesConstraint newMatchesConstraint(Matches constraint) {
114     return new DefaultMatchesConstraint(
115         toId(constraint.id()),
116         constraint.level(),
117         toMetapath(constraint.target()),
118         toPattern(constraint.pattern()),
119         toDataType(constraint.typeAdapter()),
120         toRemarks(constraint.remarks()));
121   }
122 
123   static List<DefaultKeyField> toKeyFields(KeyField... keyFields) {
124     List<DefaultKeyField> retval;
125     if (keyFields == null || keyFields.length == 0) {
126       retval = Collections.emptyList();
127     } else {
128       retval = new ArrayList<>(keyFields.length);
129       for (KeyField keyField : keyFields) {
130         DefaultKeyField field = new DefaultKeyField(toMetapath(keyField.target()),
131             toPattern(keyField.pattern()), toRemarks(keyField.remarks()));
132         retval.add(field);
133       }
134       retval = Collections.unmodifiableList(retval);
135     }
136     return retval;
137   }
138 
139   static Integer toCardinality(int value) {
140     return value < 0 ? null : Integer.valueOf(value);
141   }
142 
143   static DefaultUniqueConstraint newUniqueConstraint(IsUnique constraint) {
144     return new DefaultUniqueConstraint(
145         toId(constraint.id()),
146         constraint.level(),
147         toMetapath(constraint.target()),
148         toKeyFields(constraint.keyFields()),
149         toRemarks(constraint.remarks()));
150   }
151 
152   static DefaultIndexConstraint newIndexConstraint(Index constraint) {
153     return new DefaultIndexConstraint(
154         toId(constraint.id()),
155         constraint.level(),
156         toMetapath(constraint.target()),
157         constraint.name(),
158         toKeyFields(constraint.keyFields()),
159         toRemarks(constraint.remarks()));
160   }
161 
162   static DefaultIndexHasKeyConstraint newIndexHasKeyConstraint(IndexHasKey constraint) {
163     return new DefaultIndexHasKeyConstraint(
164         toId(constraint.id()),
165         constraint.level(),
166         toMetapath(constraint.target()),
167         constraint.indexName(),
168         toKeyFields(constraint.keyFields()),
169         toRemarks(constraint.remarks()));
170   }
171 
172   static DefaultExpectConstraint newExpectConstraint(Expect constraint) {
173     return new DefaultExpectConstraint(
174         toId(constraint.id()),
175         constraint.level(),
176         toMessage(constraint.message()),
177         toMetapath(constraint.target()),
178         toMetapath(constraint.test()),
179         toRemarks(constraint.remarks()));
180   }
181 
182   static DefaultCardinalityConstraint newCardinalityConstraint(HasCardinality constraint) {
183     return new DefaultCardinalityConstraint(
184         toId(constraint.id()),
185         constraint.level(),
186         toMetapath(constraint.target()),
187         toCardinality(constraint.minOccurs()),
188         toCardinality(constraint.maxOccurs()),
189         toRemarks(constraint.remarks()));
190   }
