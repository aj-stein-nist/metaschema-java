115   /**
116    * Construct a new {@link IClassBinding} for a Java bean annotated with the {@link Assembly}
117    * annotation.
118    * 
119    * @param clazz
120    *          the Java bean class
121    * @param bindingContext
122    *          the class binding context for which this class is participating
123    */
124   @SuppressWarnings("PMD")
125   protected DefaultAssemblyClassBinding(Class<?> clazz, IBindingContext bindingContext) {
126     super(clazz, bindingContext);
127     Objects.requireNonNull(clazz, "clazz");
128     if (!clazz.isAnnotationPresent(MetaschemaAssembly.class)) {
129       throw new IllegalArgumentException(
130           String.format("Class '%s' is missing the '%' annotation.", clazz.getName(),
131               MetaschemaAssembly.class.getName()));
132     }
133     this.metaschemaAssembly = clazz.getAnnotation(MetaschemaAssembly.class);
134     String namespace = ModelUtil.resolveNamespace(this.metaschemaAssembly.rootNamespace(), this, false);
135     String localName = ModelUtil.resolveLocalName(this.metaschemaAssembly.rootName(), null);
136     if (localName != null) {
137       if (namespace != null) {
138         this.xmlRootQName = new QName(namespace, localName);
139       } else {
140         this.xmlRootQName = new QName(localName);
141       }
142     } else {
143       this.xmlRootQName = null;
144     }
145   }

175   /**
176    * Collect all fields that are part of the model for this class.
177    * 
178    * @param clazz
179    *          the class
180    * @return an immutable collection of field and assembly instances
181    */
182   protected Collection<java.lang.reflect.Field> getModelInstanceFields(Class<?> clazz) {
183     java.lang.reflect.Field[] fields = clazz.getDeclaredFields();
184 
185     List<java.lang.reflect.Field> retval = new LinkedList<>();
186 
187     Class<?> superClass = clazz.getSuperclass();
188     if (superClass != null) {
189       // get instances from superclass
190       retval.addAll(getModelInstanceFields(superClass));
191     }
192 
193     for (java.lang.reflect.Field field : fields) {
194       if (!field.isAnnotationPresent(Assembly.class) && !field.isAnnotationPresent(Field.class)) {
195         // skip fields that aren't a field or assembly instance
196         continue;
197       }
198 
199       if (field.isAnnotationPresent(Ignore.class)) {
200         // skip this field, since it is ignored
201         continue;
202       }
203       retval.add(field);
204     }
205     return Collections.unmodifiableCollection(retval);
206   }

208   /**
209    * Initialize the flag instances for this class.
210    */
211   protected synchronized void initalizeModelInstances() {
212     if (this.modelInstances == null) {
213       Map<String, IBoundNamedModelInstance> modelInstances = new LinkedHashMap<>();
214       for (java.lang.reflect.Field field : getModelInstanceFields(getBoundClass())) {
215 
216         Assembly assemblyAnnotation = field.getAnnotation(Assembly.class);
217         if (assemblyAnnotation == null) {
218           Field fieldAnnotation = field.getAnnotation(Field.class);
219           if (fieldAnnotation != null) {
220             DefaultFieldProperty instance = DefaultFieldProperty.createInstance(this, field);
221             modelInstances.put(instance.getEffectiveName(), instance);
222           }
223         } else {
224           DefaultAssemblyProperty instance = DefaultAssemblyProperty.createInstance(this, field);
225           modelInstances.put(instance.getEffectiveName(), instance);
226         }
227       }
228       this.modelInstances
229           = modelInstances.isEmpty() ? Collections.emptyMap() : Collections.unmodifiableMap(modelInstances);
230     }
231   }

330   @Override
331   public Object readRoot(IXmlParsingContext context) throws XMLStreamException, BindingException, IOException {
332 
333     QName rootQName = getRootXmlQName();
334     XMLEventReader2 eventReader = context.getReader();
335 
336     if (eventReader.peek().isStartDocument()) {
337       eventReader.next();
338     }
339 
340     XmlEventUtil.skipWhitespace(eventReader);
341 
342     XmlEventUtil.assertNext(eventReader, XMLEvent.START_ELEMENT, rootQName);
343 
344     Object instance = newInstance();
345 
346     callBeforeDeserialize(instance, null);
347 
348     StartElement rootStart = eventReader.nextEvent().asStartElement();
349     readInternal(null, instance, rootStart, context);
350 
351     XmlEventUtil.consumeAndAssert(eventReader, XMLEvent.END_ELEMENT, rootQName);
352 
353     callAfterDeserialize(instance, null);
354 
355     return instance;
356   }

358   @Override
359   public Object readRoot(IJsonParsingContext context) throws BindingException, IOException {
360     String[] ignoreFieldsArray = getMetaschemaAssemblyAnnotation().ignoreRootJsonProperties();
361 
362     Set<String> ignoreRootFields;
363     if (ignoreFieldsArray == null || ignoreFieldsArray.length == 0) {
364       ignoreRootFields = Collections.emptySet();
365     } else {
366       ignoreRootFields = new HashSet<>(Arrays.asList(ignoreFieldsArray));
367     }
368 
369     String rootFieldName = getRootJsonName();
370     JsonToken token;
371     JsonParser parser = context.getReader();
372 
373     // first read the initial START_OBJECT
374     JsonUtil.consumeAndAssert(parser, JsonToken.START_OBJECT);
375 
376     Object instance = newInstance();
377 
378     callBeforeDeserialize(instance, null);
379 
380     boolean foundRoot = false;
381     while (!JsonToken.END_OBJECT.equals(token = parser.nextToken())) {
382       if (!JsonToken.FIELD_NAME.equals(token)) {
383         throw new BindingException(String.format("Expected FIELD_NAME token, found '%s'", token.toString()));
384       }
385 
386       String fieldName = parser.currentName();
387       if (fieldName.equals(rootFieldName)) {
388         foundRoot = true;
389         // process the object value, bound to the requested class
390         JsonUtil.assertAndAdvance(parser, JsonToken.FIELD_NAME);
391         JsonUtil.assertAndAdvance(parser, JsonToken.START_OBJECT);
392         readInternal(null, instance, context);
393 
394       } else if (ignoreRootFields.contains(fieldName)) {
395         // ignore the field
396         JsonUtil.skipNextValue(parser);
397       } else {
398         if (!context.getProblemHandler().handleUnknownRootProperty(instance, this, fieldName, context)) {
399           LOGGER.warn("Skipping unhandled top-level JSON field '{}'.", fieldName);
400           JsonUtil.skipNextValue(parser);
401         }
402       }
403     }
404 
405     JsonUtil.assertCurrent(parser, JsonToken.END_OBJECT);
406 
407     // advance past the end object
408     parser.nextToken();
409 
410     if (!foundRoot) {
411       throw new BindingException(String.format("Failed to find root field '%s'.", rootFieldName));
412     }
413 
414     callAfterDeserialize(instance, null);
415 
416     return instance;
417   }

419   @Override
420   protected void readBody(Object instance, StartElement start, IXmlParsingContext context)
421       throws IOException, XMLStreamException, BindingException {
422     Set<IBoundNamedModelInstance> unhandledProperties = new HashSet<>();
423     for (IBoundNamedModelInstance modelProperty : getModelInstances()) {
424       if (!modelProperty.read(instance, start, context)) {
425         unhandledProperties.add(modelProperty);
426       }
427     }
428 
429     // process all properties that did not get a value
430     for (IBoundNamedModelInstance property : unhandledProperties) {
431       // use the default value of the collector
432       property.setValue(instance, property.newPropertyCollector().getValue());
433 
434     }
435   }

437   @Override
438   public List<Object> readItem(Object parentInstance, IJsonParsingContext context)
439       throws IOException, BindingException {
440 
441     Object instance = newInstance();
442 
443     callBeforeDeserialize(instance, parentInstance);
444 
445     readInternal(parentInstance, instance, context);
446 
447     callAfterDeserialize(instance, parentInstance);
448 
449     return Collections.singletonList(instance);

452   protected void readInternal(@SuppressWarnings("unused") Object parentInstance, Object instance,
453       IJsonParsingContext context) throws BindingException, IOException {
454     JsonParser jsonParser = context.getReader();
455 
456     JsonUtil.assertCurrent(jsonParser, JsonToken.FIELD_NAME);
457 
458     IBoundFlagInstance jsonKey = getJsonKeyFlagInstance();
459     Map<String, ? extends IBoundNamedInstance> properties;
460     if (jsonKey == null) {
461       properties = getNamedInstances(null);
462     } else {
463       properties = getNamedInstances((flag) -> {
464         return !jsonKey.equals(flag);
465       });
466 
467       // if there is a json key, the first field will be the key
468       String key = jsonParser.getCurrentName();
469       // advance past the FIELD_NAME
470       jsonParser.nextToken();
471       Object value = jsonKey.readValueFromString(key);
472       jsonKey.setValue(instance, value.toString());
473 
474       // next the value will be a start object
475       JsonUtil.assertAndAdvance(jsonParser, JsonToken.START_OBJECT);
476     }
477 
478     Set<String> handledProperties = new HashSet<>();
479     while (!JsonToken.END_OBJECT.equals(jsonParser.currentToken())) {
480       String propertyName = jsonParser.getCurrentName();
481       IBoundNamedInstance property = properties.get(propertyName);
482 
483       boolean handled = false;
484       if (property != null) {
485         handled = property.read(instance, context);
486       }
487 
488       if (handled) {
489         handledProperties.add(propertyName);
490       } else {
491         if (LOGGER.isWarnEnabled()) {
492           LOGGER.warn("Unrecognized property named '{}' at '{}'", propertyName,
493               JsonUtil.toString(jsonParser.getCurrentLocation()));
494         }
495         JsonUtil.assertAndAdvance(jsonParser, JsonToken.FIELD_NAME);
496         JsonUtil.skipNextValue(jsonParser);
497       }
498     }
499 
500     // set undefined properties
501     for (Map.Entry<String, ? extends IBoundNamedInstance> entry : properties.entrySet()) {
502       if (!handledProperties.contains(entry.getKey())) {
503         // use the default value of the collector
504         IBoundNamedInstance property = entry.getValue();
505         property.setValue(instance, property.newPropertyCollector().getValue());
506       }
507     }
508 
509     if (jsonKey != null) {
510       // read the END_OBJECT for the JSON key value
511       JsonUtil.assertAndAdvance(jsonParser, JsonToken.END_OBJECT);
512     }
513   }

515   @Override
516   public void writeRoot(Object instance, IXmlWritingContext context) throws XMLStreamException, IOException {
517 
518     XMLStreamWriter2 writer = context.getWriter();
519 
520     writer.writeStartDocument("UTF-8", "1.0");
521 
522     QName rootQName = getRootXmlQName();
523 
524     NamespaceContext nsContext = writer.getNamespaceContext();
525     String prefix = nsContext.getPrefix(rootQName.getNamespaceURI());
526     if (prefix == null) {
527       prefix = "";
528     }
529 
530     writer.writeStartElement(prefix, rootQName.getLocalPart(), rootQName.getNamespaceURI());
531 
532     writeItem(instance, rootQName, context);
533 
534     writer.writeEndElement();
535   }

537   @Override
538   public void writeRoot(Object instance, IJsonWritingContext context) throws IOException {
539 
540     JsonGenerator writer = context.getWriter();
541 
542     // first read the initial START_OBJECT
543     writer.writeStartObject();
544 
545     writer.writeFieldName(getRootJsonName());
546 
547     writeInternal(instance, true, context);
548 
549     // end of root object
550     writer.writeEndObject();
551   }

553   /**
554    * Serializes the provided instance in JSON.
555    * 
556    * @param instance
557    *          the instance to serialize
558    * @param writeObjectWrapper
559    *          {@code true} if the start and end object should be written, or {@code false} otherwise
560    * @param context
561    *          the JSON writing context used to generate output
562    * @throws IOException
563    *           if an error occurs while writing to the output context
564    * @throws NullPointerException
565    *           if there is a JSON key configured and the key property's value is {@code null}
566    */
567   protected void writeInternal(Object instance, boolean writeObjectWrapper, IJsonWritingContext context)
568       throws IOException {
569     JsonGenerator writer = context.getWriter();
570 
571     if (writeObjectWrapper) {
572       writer.writeStartObject();
573     }
574 
575     IBoundFlagInstance jsonKey = getJsonKeyFlagInstance();
576     Map<String, ? extends IBoundNamedInstance> properties;
577     if (jsonKey == null) {
578       properties = getNamedInstances(null);
579     } else {
580       properties = getNamedInstances((flag) -> {
581         return !jsonKey.equals(flag);
582       });
583 
584       // if there is a json key, the first field will be the key
585       String key = jsonKey.getValueAsString(instance);
586       if (key == null) {
587         throw new IOException(new NullPointerException("Null key value"));
588       }
589       writer.writeFieldName(key);
590 
591       // next the value will be a start object
592       writer.writeStartObject();
593     }
594 
595     for (IBoundNamedInstance property : properties.values()) {
596       property.write(instance, context);
597     }
598 
599     if (jsonKey != null) {
600       // write the END_OBJECT for the JSON key value
601       writer.writeEndObject();
602     }
603 
604     if (writeObjectWrapper) {
605       writer.writeEndObject();
606     }
607   }
